class WhileClass{
	def printInWhile(): int{
		var i: int; 
		i = 0;
		while(i<4){
			writeln("In a loop!!");
			i = i + 1;
			if(i==3) then{
				writeln("3");
			}
			else {
				writeln("not 3");
			}
		}
		return -1;
	}
}

class conditionalClass{
	def checkBooleanStuff(): int{
		if(true) then{
			writeln("true");
		}
		if(true && false) then{
			writeln("true && false");
		}
		if(false) then{
			writeln("false");
		}
		if(true || false) then{
			writeln("true || false");
		}
		if(true && !false) then{
			writeln("true && !false");
		}
		if(1>4) then {
			writeln("1>4");
		}
		if(1<4) then {
			writeln("1<4");
		}

	}
