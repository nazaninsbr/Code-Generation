class Node {
    var hasNext : boolean;
    var next : Node;
    var data : int;

    def constructor() : int {
        hasNext = false;
        next = this;
        data = -1;
        return 0;
    }

    def setData(i : int) : int {
        data = i;
        return 0;
    }

    def getData() : int {
        return data;
    }

    def setNextNode(node : Node) : int {
        next = node;
        hasNext = true;
        return 0;
    }

    def unSetNext() : int {
        hasNext = false;
        return 0;
    }

    def hasNext() : boolean {
        return hasNext;
    }

    def getNext() : Node {
        return next;
    }
}
